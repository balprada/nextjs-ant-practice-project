1. login:
    req: {
        userEmail: string,
        userPassword: string
    }

*if role = "D"
    # Success response 
    res: {
        patientList: [] // Array of patients
        userInfo: {
            id: string,
            role: string,
        }
    }

    # Failure Response 
    res: {
        err: string
    }

*if role = "P"
    # Success response 
    res: {
        patientProfile: {} // object of patient-profile
        userInfo: {
            id: string,
            role: string,
        }
    }

    # Failure Response 
    res: {
        err: string
    }
